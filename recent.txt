const SCALE = 6.5; // match CSS scale

const SKILLS = [

const sphere = document.getElementById('skills-sphere');
const RADIUS = 180;

function createSphere() {
  sphere.innerHTML = '';
  const N = SKILLS.length;
  for (let i=0;i<N;i++){
    const skill = SKILLS[i];
    const tag = document.createElement('div');
    tag.className = 'tag';
    tag.textContent = skill.t;
    tag.style.fontFamily = 'Times New Roman, Times, serif';
    tag.style.fontSize = `${12 + skill.w}px`;
    tag.style.opacity = 0.95;
    tag.style.transition = 'color 0.2s ease';

    // Hover color change
    tag.addEventListener('mouseenter', ()=> tag.style.color = '#00ffff');
    tag.addEventListener('mouseleave', ()=> tag.style.color = 'white');

    const phi = Math.acos(-1 + (2*(i+0.5)/N));
    const theta = Math.PI * (1 + Math.sqrt(5)) * (i+0.5);

    const x = RADIUS * Math.cos(theta) * Math.sin(phi);
    const y = RADIUS * Math.sin(theta) * Math.sin(phi);
    const z = RADIUS * Math.cos(phi);

    tag.dataset.x = x; tag.dataset.y = y; tag.dataset.z = z;
    sphere.appendChild(tag);
  }
}

let rotX = 0, rotY = 0;
let targetX = 0, targetY = 0;
let isHovering = false;

const viewport = document.querySelector('.sphere-viewport');

// Only rotate on hover inside the sphere viewport
viewport.addEventListener('mouseenter', ()=> isHovering = true);
viewport.addEventListener('mouseleave', ()=> {
  isHovering = false;
  targetX = 0;
  targetY = 0;
});

viewport.addEventListener('mousemove', (e)=>{
  if(!isHovering) return;
  const rect = viewport.getBoundingClientRect();
  // adjust for CSS scale
  const offsetX = (e.clientX - rect.left) / SCALE;
  const offsetY = (e.clientY - rect.top) / SCALE;

  const centerX = rect.width / 2 / SCALE;
  const centerY = rect.height / 2 / SCALE;

  const dx = offsetX - centerX;
  const dy = offsetY - centerY;

  targetY = dx * 0.05;
  targetX = -dy * 0.05;
});

// Floating parameters
let floatOffset = 0;
let floatDirection = 1;

function animateSphere() {
  // Subtle floating
  floatOffset += 0.2 * floatDirection;
  if(floatOffset > 10 || floatOffset < -10) floatDirection *= -1;

  // Smooth rotation
  rotX += (targetX - rotX) * 0.08;
  rotY += (targetY - rotY) * 0.08;

  sphere.style.transform = `translate(-50%, calc(-50% + ${floatOffset}px)) rotateX(${rotX}deg) rotateY(${rotY}deg)`;

  Array.from(sphere.children).forEach(tag => {
    const x = parseFloat(tag.dataset.x);
    const y = parseFloat(tag.dataset.y);
    const z = parseFloat(tag.dataset.z);

    const ry = rotY * Math.PI / 180;
    const rx = rotX * Math.PI / 180;

    let x1 = x * Math.cos(ry) + z * Math.sin(ry);
    let z1 = -x * Math.sin(ry) + z * Math.cos(ry);
    let y1 = y * Math.cos(rx) - z1 * Math.sin(rx);
    let z2 = y * Math.sin(rx) + z1 * Math.cos(rx);

    const perspective = 600;
    const scale = (perspective + z2) / (perspective + RADIUS);
    const tx = x1 * scale;
    const ty = y1 * scale;
    const tz = z2;

    tag.style.transform = `translate3d(${tx}px, ${ty}px, ${tz}px) scale(${(parseFloat(tag.style.fontSize)/24) * scale})`;
    tag.style.zIndex = Math.round((tz + RADIUS) * 100);
    tag.style.opacity = 0.65 + (scale*0.6);
  });

  requestAnimationFrame(animateSphere);
}

createSphere();
animateSphere();

document.addEventListener("DOMContentLoaded", () => {
  const carousel = document.querySelector(".carousel");
  const cards = document.querySelectorAll(".project-card");
  const leftBtn = document.querySelector(".left-btn");
  const rightBtn = document.querySelector(".right-btn");

  let current = 0;

  function updateCarousel() {
    const totalCards = cards.length;
    const cardWidth = cards[0].offsetWidth;
    const carouselWidth = carousel.offsetWidth;
    const offset = (carouselWidth / 2) - (cardWidth / 2);
    const translateValue = -current * cardWidth + offset;

    carousel.style.transition = "transform 0.8s cubic-bezier(0.4, 0, 0.2, 1)";
    carousel.style.transform = `translateX(${translateValue}px)`;

    cards.forEach((card, i) => {
      const distance = Math.abs(i - current);
      const scale = i === current ? 1 : 0.8;
      const blur = i === current ? "blur(0px)" : "blur(3px)";
      const brightness = i === current ? "brightness(1)" : "brightness(0.5)";
      const opacity = i === current ? 1 : 0.5;
      const zIndex = i === current ? 10 : 1;

      card.style.transition = "all 0.6s ease";
      card.style.transform = `scale(${scale})`;
      card.style.filter = `${blur} ${brightness}`;
      card.style.opacity = opacity;
      card.style.zIndex = zIndex;
      card.classList.toggle("active", i === current);

      // === GIF PLAY CONTROL ===
      const img = card.querySelector("img");
      if (!img) return;

      const gif = img.getAttribute("data-gif");
      const still = img.getAttribute("data-still");

      if (i === current && gif) {
        // Add a unique query parameter to force reload the GIF (play)
        img.src = `${gif}?t=${new Date().getTime()}`;
      } else if (still) {
        // Show still image when not active
        img.src = still;
      }
    });
  }

  // Button navigation
  leftBtn.addEventListener("click", () => {
    current = (current - 1 + cards.length) % cards.length;
    updateCarousel();
  });

  rightBtn.addEventListener("click", () => {
    current = (current + 1) % cards.length;
    updateCarousel();
  });

  updateCarousel();
  window.addEventListener("resize", updateCarousel);
});

let i = 0;
let isTag = false;
let currentTag = "";

function typeWriter() {
  if (i < fullHTML.length) {
    const char = fullHTML[i];

    if (char === "<") {
      isTag = true;
    }

    if (isTag) {
      currentTag += char;
    } else {
      element.innerHTML += char;
    }

    if (char === ">") {
      isTag = false;
      element.innerHTML += currentTag; // append full tag at once
      currentTag = "";
    }

    i++;
    setTimeout(typeWriter, 15); // speed
  } else {
    // Remove cursor when done
    element.style.borderRight = "none";
  }
}

function validateForm(event) {
  event.preventDefault();

  const name = document.getElementById("name");
  const email = document.getElementById("email");
  const subject = document.getElementById("subject");
  const message = document.getElementById("message");

  let isValid = true;
  let errorMsg = "";

  // Reset previous styles
  [name, email, subject, message].forEach(field => field.style.borderColor = "");

  // Name validation
  if (name.value.trim().length < 2) {
    errorMsg = "Please enter your full name.";
    name.style.borderColor = "red";
    isValid = false;
  }

  // Email validation
  const emailPattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/;
  if (!email.value.match(emailPattern)) {
    errorMsg = "Please enter a valid email address.";
    email.style.borderColor = "red";
    isValid = false;
  }

  // Subject validation
  if (subject.value.trim().length < 3) {
    errorMsg = "Please enter a valid subject.";
    subject.style.borderColor = "red";
    isValid = false;
  }

  // Message validation
  if (message.value.trim().length < 10) {
    errorMsg = "Please enter at least 10 characters in your message.";
    message.style.borderColor = "red";
    isValid = false;
  }

  if (!isValid) {
    alert("‚ö†Ô∏è " + errorMsg);
    return false;
  }

  // If all fields valid, submit to Formspree
  fetch(event.target.action, {
    method: "POST",
    body: new FormData(event.target),
    headers: { 'Accept': 'application/json' }
  }).then(response => {
    if (response.ok) {
      alert("‚úÖ Message sent successfully! I‚Äôll get back to you soon.");
      event.target.reset();
    } else {
      alert("‚ùå Oops! Something went wrong. Please try again later.");
    }
  }).catch(() => {
    alert("‚ùå Network error. Please check your connection.");
  });

  return false;
}
</script>
